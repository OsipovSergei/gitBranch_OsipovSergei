1.git branch "имя ветки" создание ветки
2.git checkout -b "имя ветки" создание и переход в ветку
3.git checkout "имя ветки" переключение между ветками
4.git log отображение id коммита
5.git checkout "id коммита" переход на определённый коммит
6.git commit --amend -m "новое название коммита" изменение текста коммита
7.git push --set-upstream origin "имя ветки" пуш локальной ветки и создание её на Github (необходимо находиться в созданной ветке)
8.git restore "имя файла" возвращение состояния файла к последнему коммиту (git ver. > 2.23)
